SELECT
Persona.Nombre,
Persona.Apellidos,
Persona.Telefono,
Persona.Direccion,
Persona.Nacionalidad
FROM Persona
INNER JOIN Reserva
ON
Persona.Id=Reserva.Id_huesped
INNER JOIN Cama
ON
Reserva.Id_cama=Cama.Id
WHERE
 Cama.Id_habitacion=2
AND
 Reserva.fecha_de_salida IS NULL;
